ecewo provides a middleware feature, which looks like Express.js. Let's see how they work.

## Route Specific Middleware

```c
#include "ecewo.h"
#include <stdio.h>

void admin_only(ecewo_request_t *req, ecewo_response_t *res, Next next) {
  const char *role = ecewo_header_get(req, "X-User-Role");

  if (!role || strcmp(role, "admin") != 0) {
    ecewo_send_text(res, 403, "Admin access required");
    return;
  }

  // Continue to next middleware/handler
  next(req, res);
}

void validate_settings(ecewo_request_t *req, ecewo_response_t *res, Next next) {
  // Some validation process...
  // Continue to next middleware/handler
  next(req, res);
}

int main(void) {
  // Server setup...

  // Apply middleware to specific route
  ECEWO_GET(app, "/admin/users", admin_only, users_handler);
  ECEWO_POST(app, "/admin/settings", admin_only, validate_settings, settings_handler);

  // ... rest of setup
}
```

A request timeout example:

```c
void custom_timeout_middleware(ecewo_request_t *req, ecewo_response_t *res, Next next) {
  const char *path = req->path;

  // Different timeouts for different routes
  if (strncmp(path, "/api/", 5) == 0) {
    ecewo_timeout_request(res, 10000);  // 10s
  } else if (strncmp(path, "/upload", 7) == 0) {
    ecewo_timeout_request(res, 300000); // 5min
  }
  // else: no timeout

  next(req, res);
}

int main(void) {
  ecewo_app_t *app = ecewo_create();

  ECEWO_GET(app, "/path", custom_timeout_middleware, handler);

  if (ecewo_listen(app, 3000) != 0) {
    fprintf(stderr, "Failed to start server\n");
    return 1;
  }

  return 0;
}
```

## Global Middleware

We have `ECEWO_USE()` to define global middlewares. Let's implement a simple logger in middlewares.c and apply it before every handler.

```c
#include "ecewo.h"
#include "stdio.h"

void auth_middleware(ecewo_request_t *req, ecewo_response_t *res, Next next) {
  const char *auth = ecewo_header_get(req, "Authorization");

  if (!auth || strncmp(auth, "Bearer ", 7) != 0) {
    ecewo_send_text(res, 401, "Unauthorized");
    return;
  }

  // Continue to next middleware/handler
  next(req, res);
}

void logging_middleware(ecewo_request_t *req, ecewo_response_t *res, Next next) {
  printf("%s %s\n", req->method, req->path);
  next(req, res);
}

int main(void) {
  // Server setup ...

  // Apply to all routes
  ECEWO_USE(app, logging_middleware);
  ECEWO_USE(app, auth_middleware);

  ECEWO_GET(app, "/protected", protected_handler);
  // ... rest of setup
}
```

A global request timeout example:

```c
void timeout_middleware(ecewo_request_t *req, ecewo_response_t *res, Next next) {
  ecewo_timeout_request(res, 15000);
  next(req, res);
}

int main(void) {
  ecewo_app_t *app = ecewo_create();
  if (!app) {
    fprintf(stderr, "Failed to initialize server\n");
    return 1;
  }

  ECEWO_USE(app, timeout_middleware);

  ECEWO_GET(app, "/path1", handler1);
  ECEWO_GET(app, "/path2", handler2);
  ECEWO_GET(app, "/path3", handler3);

  if (ecewo_listen(app, 3000) != 0) {
    fprintf(stderr, "Failed to start server\n");
    return 1;
  }

  return 0;
}
```

## Path Prefix Middleware

`ECEWO_USE()` also accepts an optional path prefix as its second argument. The middleware will only run for requests whose path starts with that prefix - similar to Express.js's `app.use('/prefix', fn)`.

```c
void api_only_middleware(ecewo_request_t *req, ecewo_response_t *res, Next next) {
  const char *auth = ecewo_header_get(req, "Authorization");

  if (!auth) {
    ecewo_send_text(res, 401, "Unauthorized");
    return;
  }

  next(req, res);
}

int main(void) {
  ecewo_app_t *app = ecewo_create();
  if (!app) {
    fprintf(stderr, "Failed to initialize server\n");
    return 1;
  }

  // Only runs for /api, /api/users, /api/anything, etc.
  ECEWO_USE(app, "/api", api_only_middleware);

  ECEWO_GET(app, "/api/users", users_handler);   // middleware runs
  ECEWO_GET(app, "/api/posts", posts_handler);   // middleware runs
  ECEWO_GET(app, "/public", public_handler);     // middleware does NOT run

  if (ecewo_listen(app, 3000) != 0) {
    fprintf(stderr, "Failed to start server\n");
    return 1;
  }

  return 0;
}
```

Prefix matching rules:

- `ECEWO_USE(app, "/api", fn)` matches `/api`, `/api/users`, `/api/v2/items`, ...
- `ECEWO_USE(app, "/api", fn)` does **not** match `/apiv2` or `/api-old`
- `ECEWO_USE(app, "/", fn)` matches every path (equivalent to `ECEWO_USE(app, fn)`)
- `ECEWO_USE(app, fn)` (no path) is global - runs for all requests

## Middleware Context

There is a specific way to pass the data along the middleware chain: `ecewo_context_get()` and `ecewo_context_set()` functions. We can pass the data to the next middleware or to the handler using them.

```c
typedef struct {
  char *user_id;
  char *role;
} user_ctx_t;

void context_middleware(ecewo_request_t *req, ecewo_response_t *res, Next next) {
  const char *token = ecewo_header_get(req, "Authorization");

  // Allocate context from request arena
  user_ctx_t *ctx = ecewo_alloc(req->arena, sizeof(user_ctx_t));
  ctx->user_id = ecewo_strdup(req->arena, "user123");
  ctx->role = ecewo_strdup(req->arena, "admin");

  // Store context pointer in request
  ecewo_context_set(req, "user_ctx", ctx);

  // Continue to next middleware
  next(req, res);
}

void protected_handler(ecewo_request_t *req, ecewo_response_t *res) {
  // Retrieve context from request
  user_ctx_t *ctx = (user_ctx_t *)ecewo_context_get(req, "user_ctx");

  if (!ctx) {
    ecewo_send_text(res, ECEWO_UNAUTHORIZED, "Unauthorized");
    return;
  }

  if (strcmp(ctx->user_id, "user123") != 0 ||
    strcmp(ctx->role, "admin") != 0) {
    ecewo_send_text(res, ECEWO_FORBIDDEN, "Forbidden");
    return;
  }

  ecewo_send_text(res, ECEWO_OK, "Success!");
}
```

### Important Notes

**Context Lifecycle:**
- Context data is stored in the request arena
- Automatically cleaned up after response is sent
- Valid throughout the entire middleware chain and handler

**Memory Management:**
- Always allocate context data from `req->arena`
- `ecewo_context_set()` stores the pointer directly (no copy)
- All middleware and handlers share the same pointer

> [!NOTE]
>
> ecewo has its own arena allocator. So `ecewo_alloc()` and `ecewo_strdup()` functions are parts of it. See the [next chapter](docs/06.memory-management.md).
