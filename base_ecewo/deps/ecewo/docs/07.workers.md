# Workers

Blocking computations should not run on the main event loop as they would block all other requests. ecewo provides `ecewo_spawn()` to handle blocking operations safely by executing them in [libuv](https://libuv.org/)'s thread pool.

## Fire and Forget

The `ecewo_spawn()` function executes blocking work in a worker thread and runs a callback on the main thread when complete. Pass `res=NULL` for tasks that do not need to send an HTTP response.

**Signature:**
```c
int ecewo_spawn(ecewo_response_t *res, void *context, ecewo_spawn_handler_t work_fn, ecewo_spawn_done_t done_fn);
```

**Parameters:**
- `res`: Pass `NULL` for fire-and-forget background tasks
- `context`: User context pointer (must contain arena for memory allocation)
- `work_fn`: Function to execute in worker thread (safe to block here)
- `done_fn`: Callback when work completes (runs on main thread, can be NULL)

**Returns:**
- `0` on success
- `-1` on failure

**Callback Signatures:**
```c
typedef void (*ecewo_spawn_handler_t)(void *context);
typedef void (*ecewo_spawn_done_t)(ecewo_response_t *res, void *context);
```

### Basic Usage

```c
#include "ecewo.h"
#include <stdio.h>

typedef struct {
  ecewo_arena_t *arena;
  int iterations;
} ComputeContext;

// This runs in a worker thread (safe to block)
void calculate_primes(void *context) {
  ComputeContext *ctx = (ComputeContext *)context;

  printf("Computing primes up to %d...\n", ctx->iterations);

  int prime_count = 0;

  // CPU-intensive loop
  for (int num = 2; num <= ctx->iterations; num++) {
    int is_prime = 1;

    for (int i = 2; i * i <= num; i++) {
      if (num % i == 0) {
        is_prime = 0;
        break;
      }
    }

    if (is_prime)
      prime_count++;
  }

  printf("Found %d primes\n", prime_count);

  ecewo_arena_return(ctx->arena);
}

void compute_handler(ecewo_request_t *req, ecewo_response_t *res) {
  const char *iter_str = ecewo_query(req, "iterations");
  int iterations = atoi(iter_str);

  // Create independent arena for background work
  ecewo_arena_t *bg_arena = ecewo_arena_borrow();

  ComputeContext *ctx = ecewo_alloc(bg_arena, sizeof(ComputeContext));
  ctx->arena = bg_arena;
  ctx->iterations = iterations;

  // Offload to worker thread
  ecewo_spawn(NULL, ctx, calculate_primes, NULL);

  // Handler returns immediately
  // Computation will work on background
  char *msg = ecewo_sprintf(req->arena, "Computation started");
  ecewo_send_text(res, ECEWO_ACCEPTED, msg);
}

int main(void) {
  ecewo_app_t *app = ecewo_create();
  if (!app) {
    fprintf(stderr, "Failed to initialize server\n");
    return 1;
  }

  ECEWO_GET(app, "/compute", compute_handler);

  if (ecewo_listen(app, 3000) != 0) {
    fprintf(stderr, "Failed to start server\n");
    return 1;
  }

  return 0;
}
```

Go to `http://localhost:3000/compute?iterations=10000` address.

### Advanced Example

Here's a more complex example with multiple parallel tasks:

```c
#include "ecewo.h"

typedef struct {
  ecewo_arena_t *arena;
  int total;
  int completed;
  int results[3];
} ParallelContext;

static void parallel_work_1(void *context) {
  ParallelContext *ctx = (ParallelContext *)context;

  // CPU-intensive work
  int sum = 0;
  for (int i = 0; i < 100000000; i++) {
    sum += i % 100;
  }

  ctx->results[0] = sum % 1000;
}

static void parallel_work_2(void *context) {
  ParallelContext *ctx = (ParallelContext *)context;

  int product = 1;
  for (int i = 1; i < 1000000; i++) {
    product = (product * i) % 10000;
  }

  ctx->results[1] = product % 1000;
}

static void parallel_work_3(void *context) {
  ParallelContext *ctx = (ParallelContext *)context;

  int count = 0;
  for (int i = 2; i < 100000; i++) {
    int is_prime = 1;
    for (int j = 2; j * j <= i; j++) {
      if (i % j == 0) {
        is_prime = 0;
        break;
      }
    }
    if (is_prime)
      count++;
  }

  ctx->results[2] = count % 1000;
}

static void parallel_done(ecewo_response_t *res, void *context) {
  (void)res;
  ParallelContext *ctx = (ParallelContext *)context;
  ctx->completed++;

  // All tasks completed
  if (ctx->completed == ctx->total) {
    int sum = ctx->results[0] + ctx->results[1] + ctx->results[2];

    printf("Background calculation completed. Sum: %d\n", sum);
    printf("  Task 1: %d\n", ctx->results[0]);
    printf("  Task 2: %d\n", ctx->results[1]);
    printf("  Task 3: %d\n", ctx->results[2]);

    ecewo_arena_return(ctx->arena);
  }
}

void parallel_handler(ecewo_request_t *req, ecewo_response_t *res) {
  ecewo_arena_t *bg_arena = ecewo_arena_borrow();

  ParallelContext *ctx = ecewo_alloc(bg_arena, sizeof(ParallelContext));
  ctx->arena = bg_arena;
  ctx->total = 3;
  ctx->completed = 0;
  ctx->results[0] = 0;
  ctx->results[1] = 0;
  ctx->results[2] = 0;

  // Launch 3 parallel tasks
  ecewo_spawn(NULL, ctx, parallel_work_1, parallel_done);
  ecewo_spawn(NULL, ctx, parallel_work_2, parallel_done);
  ecewo_spawn(NULL, ctx, parallel_work_3, parallel_done);

  // Send response immediately
  // Computation will work on background
  ecewo_send_text(res, ECEWO_ACCEPTED, "Calculation started in background");
}
```

## Wait and Respond

To send a response after the blocking operation completes, pass the handler's `res` to `ecewo_spawn()`. The `done_fn` receives it back on the main thread, where it is safe to call `ecewo_send*`.

It is also usable for blocking database queries, such as SQLite. See the [SQLite Example](./examples/using_sqlite.md).

### Basic Usage

```c
#include "ecewo.h"

typedef struct {
  int result;
} ComputeContext;

// This runs in a worker thread (safe to block)
void heavy_computation(void *context) {
  ComputeContext *ctx = (ComputeContext *)context;

  // Simulate CPU-intensive work
  int sum = 0;
  for (int i = 0; i < 1000000000; i++) {
    sum += i;
  }

  ctx->result = sum;
}

// This runs on the main thread when heavy_computation is done
void computation_complete(ecewo_response_t *res, void *context) {
  ComputeContext *ctx = (ComputeContext *)context;

  char *response = ecewo_sprintf(res->arena,
                   "Computation result: %d",
                   ctx->result);

  ecewo_send_text(res, 200, response);
}

// Route handler (runs on main thread)
void compute_handler(ecewo_request_t *req, ecewo_response_t *res) {
  // Create context (arena-allocated)
  ComputeContext *ctx = ecewo_alloc(res->arena, sizeof(ComputeContext));
  ctx->result = 0;

  // Spawn the work
  ecewo_spawn(res, ctx, heavy_computation, computation_complete);

  // Handler returns immediately
  // Response is sent when computation_complete runs
}

int main(void) {
  ecewo_app_t *app = ecewo_create();
  if (!app) {
    fprintf(stderr, "Failed to initialize server\n");
    return 1;
  }

  ECEWO_GET(app, "/compute", compute_handler);

  if (ecewo_listen(app, 3000) != 0) {
    fprintf(stderr, "Failed to start server\n");
    return 1;
  }

  return 0;
}
```

### Advanced Example

```c
#include "ecewo.h"

typedef struct {
  int total;
  int completed;
  int results[3];
  bool has_error;
} ParallelContext;

static void parallel_work_1(void *context) {
  ParallelContext *ctx = (ParallelContext *)context;

  // Simulate CPU-intensive work
  // In real code: database query, file I/O, API call, etc.
  ctx->results[0] = 10;
}

static void parallel_work_2(void *context) {
  ParallelContext *ctx = (ParallelContext *)context;
  ctx->results[1] = 20;
}

static void parallel_work_3(void *context) {
  ParallelContext *ctx = (ParallelContext *)context;
  ctx->results[2] = 30;
}

static void parallel_done(ecewo_response_t *res, void *context) {
  ParallelContext *ctx = (ParallelContext *)context;
  ctx->completed++;

  // If any spawn failed
  if (ctx->has_error && ctx->completed == 1) {
    ecewo_send_text(res, 500, "Task spawn failed");
    return;
  }

  // All tasks completed successfully
  if (ctx->completed == ctx->total && !ctx->has_error) {
    int sum = ctx->results[0] + ctx->results[1] + ctx->results[2];
    char *response = ecewo_sprintf(res->arena, "{\"sum\":%d}", sum);
    ecewo_send_json(res, 200, response);
  }
}

void parallel_handler(ecewo_request_t *req, ecewo_response_t *res) {
  ParallelContext *ctx = ecewo_alloc(res->arena, sizeof(ParallelContext));
  ctx->total = 3;
  ctx->completed = 0;
  ctx->results[0] = 0;
  ctx->results[1] = 0;
  ctx->results[2] = 0;
  ctx->has_error = false;

  // Launch 3 parallel tasks
  if (ecewo_spawn(res, ctx, parallel_work_1, parallel_done) != 0)
    ctx->has_error = true;
  if (ecewo_spawn(res, ctx, parallel_work_2, parallel_done) != 0)
    ctx->has_error = true;
  if (ecewo_spawn(res, ctx, parallel_work_3, parallel_done) != 0)
    ctx->has_error = true;

  // If all spawns failed
  if (ctx->has_error && ctx->completed == 0)
    ecewo_send_text(res, 500, "Failed to spawn tasks");
}

// ===== ROUTE REGISTRATION =====

int main(void) {
  ecewo_app_t *app = ecewo_create();
  if (!app) {
    fprintf(stderr, "Failed to initialize server\n");
    return 1;
  }

  ECEWO_GET(app, "/parallel", parallel_handler);

  if (ecewo_listen(app, 3000) != 0) {
    fprintf(stderr, "Failed to start server\n");
    return 1;
  }

  return 0;
}
```

## Memory Management in Workers

### Context allocation

**`ecewo_spawn(NULL, ...)` (fire and forget):** The handler's `req->arena` is reset as soon as the response is sent, which happens before the background work finishes. Borrow a dedicated arena instead, allocate everything from it, and return it in the done callback:

```c
typedef struct {
  ecewo_arena_t *arena;
  char *input;
  char *output;
} MyContext;

void my_work(void *ctx_ptr) {
  MyContext *ctx = ctx_ptr;
  // allocate output here, once the size is known
  ctx->output = ecewo_sprintf(ctx->arena, "processed: %s", ctx->input);
}

void my_done(ecewo_response_t *res, void *ctx_ptr) {
  (void)res;
  MyContext *ctx = ctx_ptr;
  printf("result: %s\n", ctx->output);
  ecewo_arena_return(ctx->arena); // release everything at once
}

void handler(ecewo_request_t *req, ecewo_response_t *res) {
  ecewo_arena_t *arena = ecewo_arena_borrow();
  MyContext *ctx = ecewo_alloc(arena, sizeof(MyContext));
  ctx->arena = arena;
  ctx->input = ecewo_strdup(arena, "some input");
  ctx->output = NULL;
  ecewo_spawn(NULL, ctx, my_work, my_done);
  ecewo_send_text(res, ECEWO_ACCEPTED, "working...");
}
```

**`ecewo_spawn(res, ...)` (wait and respond):** `res->arena` stays valid until you call `ecewo_send*` in the done callback, so it is safe to allocate the context from it directly, as the examples above show.

### Arena allocations in work_fn

`ecewo_alloc()` has no locking. If two threads call it on the **same** arena simultaneously, both read and increment the same `count` field. So causes corrupted allocation, data race, undefined behavior.

> [!CAUTION]
>
> Never call `ecewo_alloc()`, `ecewo_strdup()`, or any other arena function on `req->arena` or `res->arena` inside `work_fn`. Those arenas are owned by the main event loop, which may be processing other requests at the same time. Call `ecewo_arena_borrow()` in handler and `ecewo_arena_return()` in the latest `done_fn`, which are runs on the main thread.

However, if the worker has **exclusive ownership** of an arena (one it borrowed itself) calling `ecewo_alloc()` on it from `work_fn` is safe, because no other thread touches it.

```c
typedef struct {
  ecewo_arena_t *arena;
  char *input;
  char *result;
} MyContext;

void my_work(void *ctx_ptr) {
  MyContext *ctx = ctx_ptr;
  // Safe: ctx->arena is worker-private, no one else touches it
  ctx->result = ecewo_sprintf(ctx->arena, "processed: %s", ctx->input);
}

void my_done(void *ctx_ptr) {
  MyContext *ctx = ctx_ptr;
  printf("%s\n", ctx->result);
  ecewo_arena_return(ctx->arena);
}
```

If borrowing an arena is not an option, you always can use `malloc/free` instead:

```c
typedef struct {
  char *result; // malloc'd by worker, freed in done
} Ctx;

void work(void *ctx_ptr) {
  Ctx *ctx = ctx_ptr;
  ctx->result = malloc(some_dynamic_size);
  // fill ctx->result ...
}

void done(void *ctx_ptr) {
  Ctx *ctx = ctx_ptr;
  printf("%s\n", ctx->result);
  free(ctx->result);
  free(ctx);
}
```

## Notes

> [!IMPORTANT]
>
> The `spawn()` function uses libuv's thread pool (default: 4 threads). All blocking operations share this pool, so tasks may wait if all threads are busy. Set `UV_THREADPOOL_SIZE` environment variable to increase the thread pool size if needed.

```c
// If you have many concurrent requests with spawn(),
// increase thread pool size before starting server:

int main(void) {
  // Set thread pool size (default is 4)
  setenv("UV_THREADPOOL_SIZE", "16", 1);

  ecewo_app_t *app = ecewo_create();
  if (!app) {
    fprintf(stderr, "Failed to initialize server\n");
    return 1;
  }

  ECEWO_GET(app, "/task", task_handler);

  if (ecewo_listen(app, 3000) != 0) {
    fprintf(stderr, "Failed to start server\n");
    return 1;
  }

  return 0;
}
```

> [!CAUTION]
>
> Never send response in work function.

```c
// WRONG - work_fn runs in worker thread
void wrong_work(void *context) {
  MyContext *ctx = (MyContext *)context;
  ecewo_send_text(ctx->res, 200, "Done"); // BUG!
}

// CORRECT - send response in done_fn
void correct_done(ecewo_response_t *res, void *context) {
  (void)context;
  ecewo_send_text(res, 200, "Done"); // Runs on main thread
}
```

> [!CAUTION]
>
> If you want to set a route-specific timeout using `ecewo_timeout_request()` in an async handler, you must call it before calling `ecewo_spawn()`. DO NOT call `ecewo_timeout_request()` in a worker!

## Usage in Middleware

```c
#include "ecewo.h"

typedef struct {
  ecewo_request_t *req,
  Next next,
  const char *token;
  bool valid;
} auth_context_t;

void auth_middleware(ecewo_request_t *req, ecewo_response_t *res, Next next) {
  const char *token = ecewo_header_get(req, "Authorization");

  if (!token) {
    ecewo_send_text(res, 401, "ERROR: No token");
    return;
  }

  auth_context_t *ctx = ecewo_alloc(req->arena, sizeof(auth_context_t));
  ctx->req = req;
  ctx->next = next;
  ctx->token = ecewo_strdup(req->arena, token);

  ecewo_spawn(res, ctx, verify_token_work, verify_token_done);
  // Do not call next() here, async process continues
}

void verify_token_work(void *context) {
  auth_context_t *ctx = (auth_context_t *)context;

  // do something...

  ctx->valid = true;
}

void verify_token_done(ecewo_response_t *res, void *context) {
  auth_context_t *ctx = (auth_context_t *)context;

  if (ctx->valid) {
    ctx->next(ctx->req, res);
  } else {
    ecewo_send_text(res, 401, "ERROR: Invalid token");
  }
}
```
